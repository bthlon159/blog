                                                    # 9.1.2 Swing дёӯзҡ„зәҝзЁӢе°Ғй—ӯжңәеҲ¶

                                                    жүҖжңү Swing з»„д»¶пјҲдҫӢеҰӮ JButton е’Ң JTableпјүе’Ңж•°жҚ®жЁЎеһӢеҜ№иұЎпјҲдҫӢеҰӮ TableModel е’Ң TreeModelпјүйғҪиў«е°Ғй—ӯеңЁдәӢд»¶зәҝзЁӢдёӯпјҢеӣ жӯӨд»»дҪ•и®ҝй—®е®ғд»¬зҡ„д»Јз ҒйғҪеҝ…йЎ»еңЁдәӢд»¶зәҝзЁӢдёӯиҝҗиЎҢгҖӮGUI еҜ№иұЎе№¶йқһйҖҡиҝҮеҗҢжӯҘжқҘзЎ®дҝқдёҖиҮҙжҖ§пјҢиҖҢжҳҜйҖҡиҝҮзәҝзЁӢе°Ғй—ӯжңәеҲ¶гҖӮиҝҷз§Қж–№жі•зҡ„еҘҪеӨ„еңЁдәҺпјҢеҪ“и®ҝй—®иЎЁзҺ°еҜ№иұЎпјҲPresentation Objectпјүж—¶еңЁдәӢд»¶зәҝзЁӢдёӯиҝҗиЎҢзҡ„д»»еҠЎж— йЎ»жӢ…еҝғеҗҢжӯҘй—®йўҳпјҢиҖҢеқҸеӨ„еңЁдәҺпјҢж— жі•д»ҺдәӢд»¶зәҝзЁӢд№ӢеӨ–зҡ„зәҝзЁӢдёӯи®ҝй—®иЎЁзҺ°еҜ№иұЎгҖӮ

                                                    Swing зҡ„еҚ•зәҝзЁӢи§„еҲҷжҳҜпјҡSwing дёӯзҡ„з»„д»¶д»ҘеҸҠжЁЎеһӢеҸӘиғҪеңЁиҝҷдёӘдәӢд»¶еҲҶеҸ‘зәҝзЁӢдёӯиҝӣиЎҢеҲӣе»әгҖҒдҝ®ж”№д»ҘеҸҠжҹҘиҜўгҖӮ

                                                    дёҺжүҖжңүзҡ„и§„еҲҷзӣёеҗҢпјҢиҝҷдёӘи§„еҲҷд№ҹеӯҳеңЁдёҖдәӣдҫӢеӨ–жғ…еҶөгҖӮSwing дёӯеҸӘжңүе°‘ж•°ж–№жі•еҸҜд»Ҙе®үе…Ёең°д»Һе…¶д»–зәҝзЁӢдёӯи°ғз”ЁпјҢиҖҢеңЁ Javadoc дёӯе·Із»ҸеҫҲжё…жҘҡең°иҜҙжҳҺдәҶиҝҷдәӣж–№жі•зҡ„зәҝзЁӢе®үе…ЁжҖ§гҖӮеҚ•зәҝзЁӢи§„еҲҷзҡ„е…¶д»–дёҖдәӣдҫӢеӨ–жғ…еҶөеҢ…жӢ¬пјҡ

                                                    • SwingUtilities.isEventDispatchThreadпјҢз”ЁдәҺеҲӨж–ӯеҪ“еүҚзәҝзЁӢжҳҜеҗҰжҳҜдәӢд»¶зәҝзЁӢгҖӮ

                                                    • SwingUtilities.invokeLaterпјҢиҜҘж–№жі•еҸҜд»Ҙе°ҶдёҖдёӘ Runnable д»»еҠЎи°ғеәҰеҲ°дәӢд»¶зәҝзЁӢдёӯжү§иЎҢпјҲеҸҜд»Ҙд»Һд»»ж„ҸзәҝзЁӢдёӯи°ғз”ЁпјүгҖӮ

                                                    • SwingUtilities.invokeAndWaitпјҢиҜҘж–№жі•еҸҜд»Ҙе°ҶдёҖдёӘ Runnable д»»еҠЎи°ғеәҰеҲ°дәӢд»¶зәҝзЁӢдёӯжү§иЎҢпјҢе№¶йҳ»еЎһеҪ“еүҚзәҝзЁӢзӣҙеҲ°д»»еҠЎе®ҢжҲҗпјҲеҸӘиғҪд»Һйқһ GUI зәҝзЁӢдёӯи°ғз”ЁпјүгҖӮ

                                                    • жүҖжңүе°ҶйҮҚз»ҳпјҲRepaintпјүиҜ·жұӮжҲ–йҮҚз”ҹж•ҲпјҲRevalidationпјүиҜ·жұӮжҸ’е…ҘйҳҹеҲ—зҡ„ж–№жі•пјҲеҸҜд»Һд»»ж„ҸзәҝзЁӢдёӯи°ғз”ЁпјүгҖӮ

                                                    • жүҖжңүж·»еҠ жҲ–з§»йҷӨзӣ‘еҗ¬еҷЁзҡ„ж–№жі•пјҲиҝҷдәӣж–№жі•еҸҜд»Ҙд»Һд»»ж„ҸзәҝзЁӢдёӯи°ғз”ЁпјҢдҪҶзӣ‘еҗ¬еҷЁжң¬иә«дёҖе®ҡиҰҒеңЁдәӢд»¶зәҝзЁӢдёӯи°ғз”ЁпјүгҖӮ

                                                    invokeLater е’Ң invokeAndWait дёӨдёӘж–№жі•зҡ„дҪңз”Ёй…·дјј ExecutorгҖӮдәӢе®һдёҠпјҢз”ЁеҚ•зәҝзЁӢзҡ„ Executor жқҘе®һзҺ° SwingUtilities дёӯдёҺзәҝзЁӢзӣёе…ізҡ„ж–№жі•жҳҜеҫҲе®№жҳ“зҡ„пјҢеҰӮзЁӢеәҸжё…еҚ•9-1 жүҖзӨәгҖӮиҝҷе№¶йқһ SwingUtilities зҡ„зңҹе®һе®һзҺ°пјҢеӣ дёә Swing зҡ„еҮәзҺ°ж—¶й—ҙиҰҒж—©дәҺ Executor жЎҶжһ¶пјҢдҪҶеҰӮжһңзҺ°еңЁжқҘе®һзҺ° SwingпјҢжҲ–и®ёеә”иҜҘйҮҮз”Ёиҝҷз§Қе®һзҺ°ж–№ејҸгҖӮ

                                                    зЁӢеәҸжё…еҚ•9-1 дҪҝз”Ё Executor жқҘе®һзҺ° SwingUtilities

                                                    public class SwingUtilities {
    private static final ExecutorService exec = Executors.newSingleThreadExecutor(new SwingThreadFactory());
    private static volatile Thread swingThread;

    private static class SwingThreadFactory implements ThreadFactory {
        public Thread newThread(Runnable r) {
            swingThread = new Thread(r);
            return swingThread;
        }
    }

    public static boolean isEventDispatchThread() {
        return Thread.currentThread() == swingThread;
    }

    public static void invokeLater(Runnable task) {
        exec.execute(task);
    }

    public static void invokeAndWait(Runnable task) throws InterruptedException, InvocationTargetException {
        Future f = exec.submit(task);
        try {
            f.get();
        } catch (ExecutionException e) {
            throw new InvocationTargetException(e);
        }
    }
}

                                                    abstract class BackgroundTask<V> implements Runnable, Future<V> {
    private final FutureTask<V> computation = new Computation();

    private class Computation extends FutureTask<V> {
        public Computation() {
            super(new Callable<V>() {
                public V call() throws Exception {
                    return BackgroundTask.this.compute();
                }
            });
        }

        protected final void done() {
            GuiExecutor.instance().execute(new Runnable() {
                public void run() {
                    V value = null;
                    Throwable thrown = null;
                    boolean cancelled = false;
                    try {
                        value = get();
                    } catch (ExecutionException e) {
                        thrown = e.getCause();
                    } catch (CancellationException e) {
                        cancelled = true;
                    } catch (InterruptedException consumed) {
                    } finally {
                        onCompletion(value, thrown, cancelled);
                    }
                }
            });
        }
    }

    protected void setProgress(final int current, final int max) {
        GuiExecutor.instance().execute(new Runnable() {
            public void run() {
                onProgress(current, max);
            }
        });
    }

    // еңЁеҗҺеҸ°зәҝзЁӢдёӯиў«еҸ–ж¶Ҳ
    protected abstract V compute() throws Exception;

    // еңЁдәӢд»¶зәҝзЁӢдёӯиў«еҸ–ж¶Ҳ
    protected void onCompletion(V result, Throwable exception, boolean cancelled) {
    }

    protected void onProgress(int current, int max) {
    }
    // Future зҡ„е…¶д»–ж–№жі•
}

                                                    дҫӢеҰӮпјҢдҪ еҸҜд»ҘдҪҝз”ЁдёҖдёӘж ‘еҪўжҺ§д»¶жқҘжҳҫзӨәиҝңзЁӢж–Үд»¶зі»з»ҹзҡ„еҶ…е®№гҖӮеңЁжҳҫзӨәж ‘еҪўжҺ§д»¶д№ӢеүҚпјҢе№¶дёҚйңҖиҰҒжһҡдёҫж•ҙдёӘж–Үд»¶зі»з»ҹвҖ”вҖ”йӮЈж ·еҒҡдјҡж¶ҲиҖ—еӨ§йҮҸзҡ„ж—¶й—ҙе’ҢеҶ…еӯҳгҖӮжӯЈзЎ®зҡ„еҒҡжі•жҳҜпјҢеҪ“ж ‘иҠӮзӮ№иў«еұ•ејҖж—¶жүҚиҜ»еҸ–зӣёеә”зҡ„еҶ…е®№гҖӮеҚідҪҝеҸӘжһҡдёҫиҝңзЁӢеҚ·дёҠзҡ„еҚ•дёӘзӣ®еҪ•д№ҹеҸҜиғҪиҠұиҙ№еҫҲй•ҝзҡ„ж—¶й—ҙпјҢеӣ жӯӨдҪ еҸҜд»ҘиҖғиҷ‘еңЁеҗҺеҸ°д»»еҠЎдёӯжү§иЎҢжһҡдёҫж“ҚдҪңгҖӮеҪ“еҗҺеҸ°д»»еҠЎе®ҢжҲҗеҗҺпјҢеҝ…йЎ»йҖҡиҝҮжҹҗз§Қж–№ејҸе°Ҷж•°жҚ®еЎ«е……еҲ°ж ‘еҪўжЁЎеһӢдёӯгҖӮеҸҜд»ҘдҪҝз”ЁзәҝзЁӢе®үе…Ёзҡ„ж ‘еҪўжЁЎеһӢжқҘе®һзҺ°иҝҷдёӘеҠҹиғҪпјҡйҖҡиҝҮ invokeLater жҸҗдәӨдёҖдёӘд»»еҠЎпјҢе°Ҷж•°жҚ®д»ҺеҗҺеҸ°д»»еҠЎдёӯ вҖңжҺЁе…ҘвҖқ дәӢд»¶зәҝзЁӢпјҢжҲ–иҖ…и®©дәӢд»¶зәҝзЁӢжұ йҖҡиҝҮиҪ®иҜўжқҘжҹҘзңӢжҳҜеҗҰжңүж•°жҚ®еҸҜз”ЁгҖӮ

                                                    # 9.4.1 зәҝзЁӢе®үе…Ёзҡ„ж•°жҚ®жЁЎеһӢ

                                                    еҸӘиҰҒйҳ»еЎһж“ҚдҪңдёҚдјҡиҝҮеәҰең°еҪұе“Қе“Қеә”жҖ§пјҢйӮЈд№ҲеӨҡдёӘзәҝзЁӢж“ҚдҪңеҗҢдёҖд»Ҫж•°жҚ®зҡ„й—®йўҳйғҪеҸҜд»ҘйҖҡиҝҮзәҝзЁӢе®үе…Ёзҡ„ж•°жҚ®жЁЎеһӢжқҘи§ЈеҶігҖӮеҰӮжһңж•°жҚ®жЁЎеһӢж”ҜжҢҒз»ҶзІ’еәҰзҡ„е№¶еҸ‘пјҢйӮЈд№ҲдәӢд»¶зәҝзЁӢе’ҢеҗҺеҸ°зәҝзЁӢе°ұиғҪе…ұдә«иҜҘж•°жҚ®жЁЎеһӢпјҢиҖҢдёҚдјҡеҸ‘з”ҹе“Қеә”жҖ§й—®йўҳгҖӮдҫӢеҰӮпјҢз¬¬ 5 з« зҡ„ DelegatingVehicleTracker еңЁеә•еұӮдҪҝз”ЁдәҶдёҖдёӘ ConcurrentHashMap жқҘжҸҗдҫӣй«ҳеәҰе№¶еҸ‘зҡ„иҜ»еҶҷж“ҚдҪңгҖӮиҝҷз§Қж–№жі•зҡ„зјәзӮ№еңЁдәҺпјҢConcurrentHashMap ж— жі•жҸҗдҫӣдёҖиҮҙзҡ„ж•°жҚ®еҝ«з…§пјҢиҖҢиҝҷеҸҜиғҪжҳҜйңҖжұӮзҡ„дёҖйғЁеҲҶгҖӮзәҝзЁӢе®үе…Ёзҡ„ж•°жҚ®жЁЎеһӢеҝ…йЎ»еңЁжӣҙж–°жЁЎжқҝж—¶дә§з”ҹдәӢд»¶пјҢиҝҷж ·и§ҶеӣҫжүҚиғҪеңЁж•°жҚ®еҸ‘з”ҹеҸҳеҢ–еҗҺиҝӣиЎҢжӣҙж–°гҖӮ

                                                    жңүж—¶еҖҷпјҢеңЁдҪҝз”ЁзүҲжң¬еҢ–ж•°жҚ®жЁЎеһӢж—¶пјҢдҫӢеҰӮ CopyOnWriteArrayList[CPJ2.2.3.3]пјҢеҸҜиғҪиҰҒеҗҢж—¶иҺ·еҫ—зәҝзЁӢе®үе…ЁжҖ§гҖҒдёҖиҮҙжҖ§д»ҘеҸҠиүҜеҘҪзҡ„е“Қеә”жҖ§гҖӮеҪ“иҺ·еҸ–дёҖдёӘ вҖңеҶҷж—¶жӢ·иҙқпјҲCopy-On-WriteпјүвҖқ е®№еҷЁзҡ„иҝӯд»ЈеҷЁж—¶пјҢиҝҷдёӘиҝӯд»ЈеҷЁе°ҶйҒҚеҺҶж•ҙдёӘе®№еҷЁгҖӮз„¶иҖҢпјҢеҸӘжңүеңЁйҒҚеҺҶж“ҚдҪңиҝңиҝңеӨҡдәҺдҝ®ж”№ж“ҚдҪңж—¶пјҢвҖңеҶҷж—¶жӢ·иҙқвҖқ е®№еҷЁжүҚиғҪжҸҗдҫӣжӣҙеҘҪзҡ„жҖ§иғҪпјҢдҫӢеҰӮеңЁиҪҰиҫҶиҝҪиёӘеә”з”ЁзЁӢеәҸдёӯе°ұдёҚйҖӮеҗҲйҮҮз”Ёиҝҷз§Қж–№жі•гҖӮдёҖдәӣзү№е®ҡзҡ„ж•°жҚ®з»“жһ„жҲ–и®ёеҸҜд»ҘйҒҝе…Қиҝҷз§ҚйҷҗеҲ¶пјҢдҪҶиҰҒжһ„е»әдёҖдёӘж—ўиғҪжҸҗдҫӣй«ҳж•Ҳзҡ„е№¶еҸ‘и®ҝй—®еҸҲиғҪеңЁж—§ж•°жҚ®ж— ж•ҲеҗҺдёҚеҶҚз»ҙжҠӨе®ғд»¬зҡ„ж•°жҚ®з»“жһ„еҚҙе№¶дёҚе®№жҳ“пјҢеӣ жӯӨеҸӘжңүе…¶д»–ж–№жі•йғҪиЎҢдёҚйҖҡеҗҺжүҚеә”иҜҘиҖғиҷ‘дҪҝз”Ёе®ғгҖӮ

                                                    # 9.4.2 еҲҶи§Јж•°жҚ®жЁЎеһӢ

                                                    еңЁеҲҶи§ЈжЁЎеһӢи®ҫи®ЎдёӯпјҢиЎЁзҺ°жЁЎеһӢиў«е°Ғй—ӯеңЁдәӢд»¶зәҝзЁӢдёӯпјҢиҖҢе…¶д»–жЁЎеһӢпјҢеҚіе…ұдә«жЁЎеһӢпјҢжҳҜзәҝзЁӢе®үе…Ёзҡ„пјҢеӣ жӯӨж—ўеҸҜд»Ҙз”ұдәӢд»¶зәҝзЁӢж–№жі•пјҢд№ҹеҸҜд»Ҙз”ұеә”з”ЁзЁӢеәҸзәҝзЁӢи®ҝй—®гҖӮиЎЁзҺ°жЁЎеһӢдјҡжіЁеҶҢе…ұдә«жЁЎеһӢзҡ„зӣ‘еҗ¬еҷЁпјҢд»ҺиҖҢеңЁжӣҙж–°ж—¶еҫ—еҲ°йҖҡзҹҘгҖӮз„¶еҗҺпјҢиЎЁзӨәжЁЎеһӢеҸҜд»ҘеңЁе…ұдә«жЁЎеһӢдёӯеҫ—еҲ°жӣҙж–°пјҡйҖҡиҝҮе°Ҷзӣёе…ізҠ¶жҖҒзҡ„еҝ«з…§еөҢе…ҘеҲ°жӣҙж–°ж¶ҲжҒҜдёӯпјҢжҲ–иҖ…з”ұиЎЁзҺ°жЁЎеһӢеңЁж”¶еҲ°жӣҙж–°дәӢд»¶ж—¶зӣҙжҺҘд»Һе…ұдә«жЁЎеһӢдёӯиҺ·еҸ–ж•°жҚ®гҖӮ
